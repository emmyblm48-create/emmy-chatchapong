function setupTopBarProfile() { if (!userData) return; try { const avatarImg = document.getElementById('top-bar-avatar'); const usernameTxt = document.getElementById('top-bar-username'); const roleTxt = document.getElementById('top-bar-user-role'); // นำข้อมูลจาก userData ไปแสดงผลบนหน้าจอ if (usernameTxt) { usernameTxt.textContent = userData.nickname || userData.username || 'Guest'; } if (roleTxt && userData.role) { roleTxt.textContent = userData.role; } // ✨ แก้ไขตรงนี้: เปลี่ยนมาดึงค่าจาก userData.profile ตามชื่อคอลัมน์ในตาราง Sheets งับ if (avatarImg && userData.profile) { avatarImg.src = userData.profile; } } catch (error) { console.error("ไม่สามารถโหลดข้อมูลโปรไฟล์สำหรับ Top Bar ได้:", error); } }
